    public Pedido getPedido(double id) throws Exception{
        Iterator it = this.getPedidos().entrySet().iterator();
        while(it.hasNext()){
            Map.Entry peds = (Map.Entry) it.next();
            Pedido p = (Pedido)peds.getValue();
            if(p.getId() == id){
                return p;
            }
        }
        throw new Exception("No existe el pedido...");
    }
